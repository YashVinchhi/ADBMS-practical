CREATE TABLE Student (
    StudentID INT PRIMARY KEY,
    StudentName VARCHAR(100),
    Age INT,
    Major VARCHAR(100),
    FacultyID INT
);

CREATE TABLE Faculty (
    FacultyID INT PRIMARY KEY,
    FacultyName VARCHAR(100),
    Department VARCHAR(100)
);

01 QUERY: Target Students matching Dr. Smith's Dept 
SELECT StudentName
FROM Student
WHERE Major = (
    SELECT Department
    FROM Faculty
    WHERE FacultyName = 'Dr. Smith'
);
02 QUERY: Identify Faculty with Youngest Student 
SELECT f.FacultyName
FROM Faculty f
JOIN Student s ON f.FacultyID = s.FacultyID
WHERE s.Age = (
    SELECT MIN(Age) FROM Student
);
03 QUERY: Match Major with Faculty ID 103 
SELECT StudentName
FROM Student
WHERE Major = (
    SELECT Department
    FROM Faculty
    WHERE FacultyID = 103
);
04 QUERY: Detect Faculty with Zero Students 
SELECT FacultyName
FROM Faculty
WHERE FacultyID NOT IN (
    SELECT DISTINCT FacultyID FROM Student
);
05 QUERY: Filter Students > Avg Age 
SELECT StudentName
FROM Student
WHERE Age > (
    SELECT AVG(Age) FROM Student
);
06 QUERY: Locate Advisor for 'Charlie' 
SELECT f.FacultyName
FROM Faculty f
JOIN Student s ON f.FacultyID = s.FacultyID
WHERE s.StudentName = 'Charlie';
07 QUERY: Students in same Dept as 'Bob' 
SELECT StudentName
FROM Student
WHERE Major = (
    SELECT Major
    FROM Student
    WHERE StudentName = 'Bob'
);
08 QUERY: Faculties with 'Chemistry' Students 
SELECT DISTINCT f.FacultyName
FROM Faculty f
JOIN Student s ON f.FacultyID = s.FacultyID
WHERE s.Major = 'Chemistry';
09 QUERY: Advised by 'Mathematics' Dept 
SELECT s.StudentName
FROM Student s
JOIN Faculty f ON s.FacultyID = f.FacultyID
WHERE f.Department = 'Mathematics';
10 QUERY: Calc Avg Age for 'Computer Science' 
SELECT AVG(Age) as AverageAge
FROM Student
